  from math import factorial

  def Pascal(numRows):
      triangle=""
      for i in range(numRows):
          for j in range(numRows-i+1):
              triangle+="  "
          # loop to get elements of row i
          for j in range(i+1):
              # nCr = n!/((n-r)!*r!)
              triangle+=( str(factorial(i)//(factorial(j)*factorial(i-j))) + "   " )

          triangle+="\n"

      return triangle
#+end_src
